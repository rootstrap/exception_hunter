def setup_routes
  if options[:users]
    gsub_file 'config/routes.rb',
              "\n  devise_for :#{plural_table_name}, skip: :all",
              "\n  ExceptionHunter.routes(self)"
  else
    route 'ExceptionHunter.routes(self)'
  end
